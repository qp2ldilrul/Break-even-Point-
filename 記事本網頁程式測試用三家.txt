    function update() {
        const fixedExpense = 4623704 / 2; // 取單月固定費用
        const margin = document.getElementById('margin').value / 100;
        const bonus = parseFloat(document.getElementById('bonus').value);
        const mLabel = document.getElementById('marginVal');
